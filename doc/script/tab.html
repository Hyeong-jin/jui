jui.ready(function(ui, uix, _) {
	tab_1 = uix.tab("#tab_1", {
		event: {
			change: function(data) {
				alert(data.text);
			},
			changeMenu: function(data) {
				alert(data.text);
			}
		},
		target: "#tab_contents_1",
		index: 2
	});
});
